/**
 * Hook genérico para autosave de drafts con debounce 500ms.
 *
 * Usage (Phase 2):
 *   useDraftAutosave('morning', getLocalDayKey(), { mood, sleep, comment });
 *
 * Behavior:
 *   - Cada cambio en `payload` reinicia el timer.
 *   - Tras 500ms sin cambios, escribe a SQLite via draftsRepository.upsert.
 *   - Cleanup en unmount cancela el timer pendiente.
 *   - NO escribe si payload no cambió (deep-equal o JSON.stringify previo).
 */
export function useDraftAutosave(
  kind: string,
  key: string,
  payload: unknown,
  options?: { debounceMs?: number },  // default 500
): void;

   import { useEffect, useRef } from 'react';
   import * as draftsRepo from '../repositories/draftsRepository';

   export function useDraftAutosave(
     kind: string,
     key: string,
     payload: unknown,
     options: { debounceMs?: number } = {},
   ): void {
     const { debounceMs = 500 } = options;
     const lastJsonRef = useRef<string | null>(null);
     const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

     useEffect(() => {
       const json = JSON.stringify(payload);
       if (json === lastJsonRef.current) return;   // no change → skip
       if (timerRef.current) clearTimeout(timerRef.current);
       timerRef.current = setTimeout(() => {
         draftsRepo.upsert(kind, key, json).catch((err) => {
           console.warn('[useDraftAutosave] upsert failed:', err instanceof Error ? err.message : 'unknown');
         });
         lastJsonRef.current = json;
       }, debounceMs);
       return () => {
         if (timerRef.current) clearTimeout(timerRef.current);
       };
     }, [kind, key, payload, debounceMs]);
   }
