  kind TEXT NOT NULL CHECK(kind IN ('quote','future')),
  text TEXT NOT NULL,
  author TEXT,
  created_at TEXT NOT NULL,
  updated_at TEXT NOT NULL

  week_key TEXT NOT NULL UNIQUE,
  answers_json TEXT NOT NULL,
  created_at TEXT NOT NULL,
  updated_at TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS drafts (
  id TEXT PRIMARY KEY,
  kind TEXT NOT NULL,
  key TEXT NOT NULL,
  payload_json TEXT NOT NULL,
  updated_at TEXT NOT NULL
);
CREATE UNIQUE INDEX IF NOT EXISTS idx_drafts_kind_key ON drafts(kind, key);

-- INSERT...SELECT con DEFAULT stamps. id preservado.
INSERT INTO mood_log (id, kind, date_key, occurred_at, mood_value, mood_scale_version, sleep_hours, comment, habit_id, created_at, updated_at)
SELECT
  id,
  'reflection',
  substr(timestamp, 1, 10),    -- date_key = YYYY-MM-DD
  timestamp,                    -- occurred_at preserves full timestamp
  value,
  'v1',
  NULL,
  description,
  habit_id,
  timestamp,                    -- created_at = original timestamp
  timestamp                     -- updated_at = original timestamp
FROM mood_entries;

   export async function cleanupPreV2Snapshots(): Promise<void> {

       const dir = FileSystem.documentDirectory;
       if (!dir) return;
